11. C++ Program to convert Binary to Decimal

Code:

        
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int binary, decimal = 0, i = 0, remainder;

    cout << "Enter a binary number: ";
    cin >> binary;

    while (binary != 0) {
        remainder = binary % 10;
        decimal += remainder * pow(2, i);
        binary /= 10;
        i++;
    }

    cout << "Decimal equivalent: " << decimal;

    return 0;
}

Output:

Enter a binary number: 1011
Decimal equivalent: 11

12. C++ Program to convert Decimal to Binary

Code:

        
#include <iostream>
using namespace std;

int main() {
    int decimal, binary[32], i = 0;
    cout << "Enter a decimal number: ";
    cin >> decimal;
    while (decimal > 0) {
        binary[i] = decimal % 2;  
        decimal = decimal / 2;
        i++;
    }
    cout << "Binary equivalent: ";
    for (int j = i - 1; j >= 0; j--) {
        cout << binary[j];
    }
    return 0;
}

Output:

Enter a decimal number: 10
Binary equivalent: 1010

14. C++ Program to convert Decimal to Hex

Code:

        
#include <iostream>
using namespace std;

int main() {
    int decimal, remainder;
    char hex[50];
    int i = 0;

    cout << "Enter a decimal number: ";
    cin >> decimal;

    while (decimal > 0) {
        remainder = decimal % 16;

        if (remainder < 10)
            hex[i] = remainder + '0';
        else
            hex[i] = remainder - 10 + 'A';

        decimal = decimal / 16;
        i++;
    }

    cout << "Hexadecimal equivalent: ";
    for (int j = i - 1; j >= 0; j--) {
        cout << hex[j];
    }

    return 0;
}

Output:

Enter a decimal number: 26
Hexadecimal equivalent: 1A

15. C++ Program to convert Octal to Decimal
