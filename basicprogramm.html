111. Write a C++ program to convert Roman number to an integer number


#include<iostream>
#include<map>
using namespace std;
int romanToInt(string s){
    map<char,int> r{{'I',1},{'V',5},{'X',10},{'L',50},{'C',100},{'D',500},{'M',1000}};
    int res=0;
    for(int i=0;i<s.size();i++){
        if(i+1<s.size() && r[s[i]]<r[s[i+1]]) res -= r[s[i]];
        else res += r[s[i]];
    }
    return res;
}
int main(){
    string s;
    cin>>s;
    cout<<"Integer: "<<romanToInt(s);
    return 0;
}

Output:

Input: IX
Integer: 9
