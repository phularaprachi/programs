3. Write a C++ program to create an abstract class Shape with abstract methods calculateArea() and calculatePerimeter(). Create derived classes Circle and Triangle that extend the Shape class and implement the respective methods to calculate the area and perimeter of each shape.

Code:

        
#include<iostream>
using namespace std;
class Shape{
public:
virtual void calculateArea()=0;
virtual void calculatePerimeter()=0;
};
class Circle:public Shape{
float r;
public:
Circle(float x){r=x;}
void calculateArea(){
cout<<"Area of circle="<<3.14*r*r<<endl;
}
void calculatePerimeter(){
cout<<"Perimeter of circle="<<2*3.14*r<<endl;
}
};
class Triangle:public Shape{
float a,b,c;
public:
Triangle(float x,float y,float z){
a=x;b=y;c=z;
}
void calculateArea(){
float s=(a+b+c)/2;
cout<<"Area of triangle="<<sqrt(s*(s-a)*(s-b)*(s-c))<<endl;
}
void calculatePerimeter(){
cout<<"Perimeter of triangle="<<a+b+c<<endl;
}
};
int main(){
Circle c(3);
Triangle t(3,4,5);
c.calculateArea();
c.calculatePerimeter();
t.calculateArea();
t.calculatePerimeter();
return 0;
}

Output:

        Area of circle=28.26
Perimeter of circle=18.84
Area of triangle=6
Perimeter of triangle=12

public:

    virtual void perimeter()=0;
};

class Triangle:public GeometricShape{
float a,b,c,h;

    Triangle(float x,float y,float z,float height){a=x;b=y;c=z;h=height;}

        cout<<"Area of triangle="<<0.5*a*h<<endl;
    }
    void perimeter(){
        cout<<"Perimeter of triangle="<<a+b+c<<endl;
    }
};

class Square:public GeometricShape{
float side;

    void area(){
        cout<<"Area of square="<<side*side<<endl;

    void perimeter(){

};

int main(){
Triangle t(3,4,5,4);
Square s(5);
t.area();
t.perimeter();
s.area();
s.perimeter();

}
