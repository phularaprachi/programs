    #include<iostream>
using namespace std;
void swap(char &a,char &b)
{
    char temp=a;
    a=b;
    b=temp;
}
void permute(char str[],int l,int r)
{
    int i;
    if(l==r)
    {
        cout<<str<<endl;
        return;
    }
    for(i=l;i<=r;i++)
    {
        swap(str[l],str[i]);
        permute(str,l+1,r);
        swap(str[l],str[i]);  
    }
}

39. Write a C++ method to check whether a string is a valid password.?

Code:


#include<iostream>
using namespace std;
int checkPassword(char str[]){
    int i=0,len=0,digit=0;
    while(str[i]!='\0'){
        len++;
        if(str[i]>='0'&&str[i]<='9')
            digit=1;
        i++;
    }
    if(len>=6&&digit==1)
        return 1;
    else
        return 0;
}

        cout<<"Valid Password";
    else
        cout<<"Invalid Password";
