if(f.is_open()){
    while(getline(f,a[n])){
    n++;
}
}
f.close();
for(i=0;i<n;i++){
    for(j=i+1;j<n;j++){
        if(a[i]>a[j]){ 
            t=a[i];
            a[i]=a[j];
            a[j]=t;
        }
    }
}

    f2<<a[i]<<endl;

15. Write a C++ program to store the information of 10 Students having rollno,name,marks,mobileno properties, get() and set() member functions as binary file with following options 1. Insert Record 2. Delete Record 3. Update Record (based on rollno) 4. Search Record (based on rollno) 5. Clone the file 6. Exit.

Code:

        
           #include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

class Student {
    int rollno;
    char name[50];
    float marks;
    char mobileno[15];

public:
    void set() {
        cout << "Enter Roll No: ";
        cin >> rollno;
        cout << "Enter Name: ";
        cin >> name;
        cout << "Enter Marks: ";
        cin >> marks;
        cout << "Enter Mobile No: ";
        cin >> mobileno;
    }

    void get() {
        cout << "\nRoll No: " << rollno;
        cout << "\nName: " << name;
        cout << "\nMarks: " << marks;
        cout << "\nMobile No: " << mobileno << endl;
    }

    int getRoll() {
        return rollno;
    }
};


void insertRecord() {
    Student s;
    ofstream file("student.dat", ios::binary | ios::app);
    s.set();
    file.write((char*)&s, sizeof(s));
    file.close();
}


void displayAll() {
    Student s;
    ifstream file("student.dat", ios::binary);
    while(file.read((char*)&s, sizeof(s))) {
        s.get();
    }
    file.close();
}

void searchRecord(int r) {
    Student s;
    ifstream file("student.dat", ios::binary);
    bool found = false;

    while(file.read((char*)&s, sizeof(s))) {
        if(s.getRoll() == r) {
            s.get();
            found = true;
        }
    }

    if(!found)
        cout << "Record not found.\n";

    file.close();
}

void deleteRecord(int r) {
    Student s;
    ifstream file("student.dat", ios::binary);
    ofstream temp("temp.dat", ios::binary);

    while(file.read((char*)&s, sizeof(s))) {
        if(s.getRoll() != r)
            temp.write((char*)&s, sizeof(s));
    }

    file.close();
    temp.close();

    remove("student.dat");
    rename("temp.dat", "student.dat");
}


void updateRecord(int r) {
    Student s;
    fstream file("student.dat", ios::binary | ios::in | ios::out);

    while(file.read((char*)&s, sizeof(s))) {
        if(s.getRoll() == r) {
            cout << "Enter new details:\n";
            s.set();

            file.seekp(-sizeof(s), ios::cur);
            file.write((char*)&s, sizeof(s));
        }
    }

    file.close();
}


void cloneFile() {
    ifstream src("student.dat", ios::binary);
    ofstream dest("clone.dat", ios::binary);

    dest << src.rdbuf();

    src.close();
    dest.close();

    cout << "File cloned successfully.\n";
}
int main() {
    int choice, roll;

    do {
        cout << "\n--- MENU ---\n";
        cout << "1. Insert Record\n";
        cout << "2. Delete Record\n";
        cout << "3. Update Record\n";
        cout << "4. Search Record\n";
        cout << "5. Clone File\n";
        cout << "6. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch(choice) {
            case 1:
                insertRecord();
                break;
            case 2:
                cout << "Enter Roll No to delete: ";
                cin >> roll;
                deleteRecord(roll);
                break;
            case 3:
                cout << "Enter Roll No to update: ";
                cin >> roll;
  updateRecord(roll);
                break;

            case 4:
                cout << "Enter Roll No to search: ";
                cin >> roll;
                searchRecord(roll);
                break;

            case 5:
                cloneFile();
                break;

            case 6:
                cout << "Exiting...\n";
                break;

            default:
                cout << "Invalid choice!\n";
        }

    } while(choice != 6);

    return 0;
}

16. Write a C++ program to calculate the average of numbers stored in a file.

Code:

        
        #include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream file("numbers.txt"); 
    int num, sum = 0, count = 0;

    if (!file) {
        cout << "Error opening file!";
        return 0;
    }
    while (file >> num) {
        sum += num;
        count++;
    }
    file.close();
    if (count != 0) {
        float avg = (float)sum / count;
        cout << "Average = " << avg;
    } else {
        cout << "File is empty!";
    }

    return 0;
}

File:

10 20 30 40 50

Output:

10 20 30 40 50
