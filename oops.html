4. Enter student details like roll name, fee, DOB and print all the detail by ascending order of roll number ?

Code:

#include <stdio.h>
struct student
{
    int roll;
    char name[20];
    int fee;
    char dob[20];
};

int main()
{
    struct student stu[5], temp;

    printf("Enter details of 5 students:\n");

    for (int i = 0; i < 5; i++)
    {

        printf("\nStudent %d:\n", i + 1);
        printf("Enterc your roll number: ");
        scanf("%d", &stu[i].roll);
        fflush(stdin);

        printf("Enter your Name: ");
        scanf("%s", stu[i].name);
        fflush(stdin);

        printf("Enter the fees: ");
        scanf("%d", stu[i].fee);
        fflush(stdin);

        printf("Enter the DOB: ");
        scanf("%s", stu[i].dob);
        fflush(stdin);
    }

    for (int i = 0; i < 5 - 1; i++)
    {
        for (int j = i + 1; j < 5; j++)
        {
            if (stu[i].roll > stu[j].roll)
            {
                temp = stu[i];
                stu[i] = stu[j];
                stu[j] = temp;
            }
        }
    }

    printf("\n-------------------------------------------------------------\n");
    printf("| %-10s | %-15s | %-10s | %-12s |\n", "Roll No", "Name", "Fee", "DOB");
    printf("-------------------------------------------------------------\n");

    for (int i = 0; i < 5; i++)
    {
        printf("| %-10d | %-15s | %-10d | %-12s |\n",
               stu[i].roll, stu[i].name, stu[i].fee, stu[i].dob);
    }

    printf("-------------------------------------------------------------\n");

    return 0;
}
